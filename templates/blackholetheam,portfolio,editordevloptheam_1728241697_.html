        function blackholeEffect(x, y, mass) {
          const G = 6.67430e-11; // Gravitational constant
          const c = 3.0e8; // Speed of light
          const r = Math.sqrt(x * x + y * y);
          const force = (G * mass) / (r * r);
          const angle = Math.atan2(y, x);
          const vx = -force * Math.cos(angle) / c;
          const vy = -force * Math.sin(angle) / c;
          return [vx, vy];
        }
