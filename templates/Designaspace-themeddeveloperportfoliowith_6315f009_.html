Project 3

A machine learning model for image classification, built with TensorFlow.js.


import * as tf from '@tensorflow/tfjs';

async function loadModel() {
  const model = await tf.loadLayersModel('https://my-model.com/model.json');
  return model;
}

async function classifyImage(model, image) {
  const predictions = await model.predict(image);
  return predictions.argMax(-1).dataSync()[0];
}
