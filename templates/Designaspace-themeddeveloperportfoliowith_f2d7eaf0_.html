Galaxy Rotator

An interactive code snippet that rotates a galaxy in the background.

        
  const galaxy = document.querySelector('.galaxy');

  function rotateGalaxy() {
    galaxy.style.transform = 'rotate(' + (performance.now() * 0.01) + 'deg)';
    requestAnimationFrame(rotateGalaxy);
  }

  rotateGalaxy();

Constellation Animator

A code snippet that creates an animated constellation representing a coding project.

        
  const constellation = document.querySelector('.constellation');
  const stars = constellation.querySelectorAll('.star');

  function animateConstellation() {
    stars.forEach(star => {
      star.style.opacity = Math.random() * 0.5 + 0.5;
      star.style.transform = 'translate(' + (Math.random() * 20 - 10) + 'px, ' + (Math.random() * 20 - 10) + 'px)';
    });
    requestAnimationFrame(animateConstellation);
  }

  animateConstellation();

Glowing Text Animator

A code snippet that creates a glowing text effect for headings and titles.

        
  const glowingText = document.querySelectorAll('.glowing-text');

  function animateGlowingText() {
    glowingText.forEach(text => {
      text.style.textShadow = '0 0 10px rgba(0, 198, 255, 0.5), 0 0 20px rgba(0, 198, 255, 0.3), 0 0 30px rgba(0, 198, 255, 0.1)';
    });
    requestAnimationFrame(animateGlowingText);
  }

  animateGlowingText();
